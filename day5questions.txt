class Solution {
public:
    int longestConsecutive(vector<int>& arr) {
        if(arr.size() == 0) return 0;

        sort(arr.begin(), arr.end());

        int count = 1;
        int maxcount = 1;

        for(int i = 0; i < arr.size() - 1; i++) {
            if(arr[i+1] == arr[i] + 1) {
                count++;
            }
            else if(arr[i+1] == arr[i]) {
                continue;              }
            else {
                count = 1;
            }
            maxcount = max(maxcount, count);
        }
        return maxcount;
    }
};
